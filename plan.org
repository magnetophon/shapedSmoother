#!/usr/bin/env bash
# Patch shapedSmoother_core.dsp: add linear fade bypass for very short attacks.
# Run from the shapedSmoother directory:
#   bash ~/Downloads/add_linear_fade.sh shapedSmoother_core.dsp
set -euo pipefail
dsp=${1:?usage: add_linear_fade.sh shapedSmoother_core.dsp}

# Verify the target line exists
if ! grep -qF 'delta = spd * stp * corrMult;' "$dsp"; then
    echo "ERROR: could not find 'delta = spd * stp * corrMult;' in $dsp" >&2
    exit 1
fi

# Replace the delta line with linear fade bypass
sed -i '/delta = spd \* stp \* corrMult;/c\
        shapedDelta = spd * stp * corrMult;\
        useLinear = int(attSamp) < oc.overshootLINEAR_BELOW;\
        linearDelta = (lhx - dx) * active * stp;\
        delta = select2(useLinear, shapedDelta, linearDelta);' "$dsp"

echo "Patched $dsp — added linear fade for att < overshootLINEAR_BELOW"
